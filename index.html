package controllers

  import "github.com/gophersaurus/gf.v1/http"

  // Home is a business  controller.
  var Home = struct {
    Index func(resp http.Responder, req *http.Request)
  }{

    // Index is an action method for a GET request.
    Index: func(resp http.Responder, req *http.Request) {
      resp.Write(req, "Welcome fellow gopher.")
    }
  }
